def getGreedyAction(Qs):
    '''
    Qs = Q table entries for state s
    '''
    return np.argmax(Qs)

def getEpsilonGreedyAction(Qs, epsilon):
    '''
    Qs = Q table entries for state s
    epsilon = parameter for exploration of non-greedy parameter
    '''
    
    nA = len(Qs)
    
    gA = getGreedyAction(Qs)
    greedyProb = 1 - epsilon + epsilon/nA
    restProb   = epsilon / nA
    
    probs = [(greedyProb if i == gA else restProb) for i in range(nA)]
    action = np.random.choice( range(nA), p = probs )
    
    return action

def sarsaUpdateQ( Qsa, Qnsa, r, alpha, gamma):
    '''
    Qsa = Q[state][action]
    Qnsa = Q[next_state][next_action]
    r = reward
    alpha = constant alpha
    gamma = discounting factor
    '''
    v = Qsa*(1 - alpha) + alpha * ( r + gamma*Qnsa )
    return v

def sarsa(env, num_episodes, alpha, gamma=1.0):
    # initialize action-value function (empty dictionary of arrays)
    Q = defaultdict(lambda: np.zeros(env.nA))
    # initialize performance monitor
    # loop over episodes
    
    gains = []
    
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()   
        
        ## TODO: complete the function
        epsilon = 1
        delEpsilon = 0.25
        minEpsilon = 1e-10
        
        state = env.reset()
        action = getEpsilonGreedyAction(Q[state], epsilon)
    
        gain = 0
        for i in range(500):
            
            if epsilon > minEpsilon:
                epsilon *= delEpsilon
                
            nState, reward, done, info = env.step(action)
            gain += reward
            
            if done:
                Q[state][action] = sarsaUpdateQ(Q[state][action], 
                                            0, reward, alpha, gamma)
                gains.append(gain)
                break
                
            nAction = getEpsilonGreedyAction(Q[nState], epsilon)
            Q[state][action] = sarsaUpdateQ(Q[state][action], 
                                            Q[nState][nAction], reward, alpha, gamma)
            
            state = nState
            action = nAction
        
    plt.figure(figsize=(15, 4))
    n = 50
    gains = np.cumsum(gains)
    gains = (gains[n:] - gains[:-n])/n
    plt.plot(gains)
    plt.show()
    plt.close('all')
        
        
    return Q

def getProb(Qs, epsilon):
    nA = len(Qs)
    
    gA = getGreedyAction(Qs)
    greedyProb = 1 - epsilon + epsilon/nA
    restProb   = epsilon / nA
    
    probs = [(greedyProb if i == gA else restProb) for i in range(nA)]
    
    return probs

def expectedQUpdateQ( Qsa, Qns, r, alpha, gamma, epsilon):
    '''
    Qsa = Q[state][action]
    Qnsa = Q[next_state][next_action]
    r = reward
    alpha = constant alpha
    gamma = discounting factor
    '''
    
    probs = getProb(Qns, epsilon)
    Qnsa  = sum(probs * Qns)
    v = Qsa*(1 - alpha) + alpha * ( r + gamma*Qnsa )
    return v
